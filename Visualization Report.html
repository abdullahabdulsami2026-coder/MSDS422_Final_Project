# Load the dataset (treating '?' as missing values)
df = pd.read_csv('diabetic_data.csv', na_values='?')

# Target Variable Transformation
# We define 'Success' as Readmission < 30 days (High Risk)
df['readmission_binary'] = df['readmitted'].apply(lambda x: 1 if x == '<30' else 0)

print(f"Dataset Shape: {df.shape}")
print(f"Baseline Readmission Rate (<30 days): {df['readmission_binary'].mean():.2%}")

plt.figure(figsize=(10, 5))
null_counts = df.isnull().sum()
null_counts = null_counts[null_counts > 0].sort_values(ascending=False)
null_counts_pct = (null_counts / len(df)) * 100

ax = null_counts_pct.plot(kind='bar', color='teal')
plt.title('Percentage of Missing Values per Feature')
plt.ylabel('Percentage (%)')
plt.xlabel('Features')

# Annotate bars
for p in ax.patches:
    ax.annotate(f'{p.get_height():.1f}%', (p.get_x() + p.get_width()/2., p.get_height()), 
                ha='center', va='center', xytext=(0, 9), textcoords='offset points')

plt.show()
